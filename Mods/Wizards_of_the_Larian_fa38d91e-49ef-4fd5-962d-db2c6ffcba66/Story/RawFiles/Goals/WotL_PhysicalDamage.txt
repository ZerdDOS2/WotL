//REGION Physical to Earth Damage
// This transforms all physical damage to earth damage for two reasons.
// One is for compatibility sake, and second is to avoid earth particle
// effects on weapons.
IF
	NRD_OnPrepareHit(_, _, _, _Handle)
AND
	NRD_HitGetDamage(_Handle, "Physical", _Damage)
THEN
	NRD_HitClearDamage(_Handle, "Physical");
	NRD_HitAddDamage(_Handle, "Earth", _Damage);
//END_REGION
EXITSECTION
