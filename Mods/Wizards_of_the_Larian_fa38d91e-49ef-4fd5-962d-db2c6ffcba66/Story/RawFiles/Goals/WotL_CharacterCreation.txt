Version 1
SubGoalCombiner SGC_AND
INITSECTION

KBSECTION
//REGION After Character Creation for Players
IF
	CharacterEnteredRegion(_Char, _)
AND
	NOT DB_CharacterCreationDummy(_Char)
AND
	CharacterIsPlayer(_Char, 1)
AND
	ObjectGetFlag(_Char, "WotL_CharacterInitialized", 0)
THEN
	ProcObjectTimer(_Char, "WotL_CharacterCreationFinished", 10000);

PROC
	ProcObjectTimerFinished((CHARACTERGUID)_Char, "WotL_CharacterCreationFinished")
THEN
	SetStoryEvent(_Char, "WotL_InitializeCharacter");
//END_REGION

//REGION Lock Abilities on CC Dummies
IF
	StoryEvent((CHARACTERGUID)_Char, "WotL_InitializeCharacter")
AND
	DB_CharacterCreationDummy(_Char)
AND
	ObjectGetFlag(_Char, "WotL_CharacterHasLockedAbilities", 0)
AND
	DB_WotL_UnusedAbilities(_Ability)
THEN
	CharacterLockAbility(_Char, _Ability);
	ProcObjectTimer(_Char, "WotL_LockedAbilitiesFlag", 1000);
//END_REGION
EXITSECTION

ENDEXITSECTION
ParentTargetEdge "__WotL_Goal"
